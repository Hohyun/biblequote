\newcommand\bverse[3]{%
  \setcounter{bverse}{#3}
  \ifthenelse{\equal{#3}{1}}{}{\vsep\marginpar{\footnotesize #3}}
  \trim@spaces{\biblequote{#1}{#2:#3}{\bversion}}%
}

\newcommand\bchapter[3]{

  \setcounter{bverse}{0}
  \lettrine{\liningnums{#2}}{}
  \foreach \i in {1,...,#3}{%
    \bverse{#1}{#2}{\i}%
  }
}
